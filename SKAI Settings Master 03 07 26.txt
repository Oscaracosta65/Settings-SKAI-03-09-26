<?php
defined('_JEXEC') or die('Restricted access');

use Joomla\CMS\Factory;
use Joomla\CMS\Session\Session;
use Joomla\Database\DatabaseInterface;

/* -----------------------------------------------------
   1) Load Lotteries from the known absolute path
   ----------------------------------------------------- */
$jsonFile = '/home/oscara/web/lottoexpert.net/public_html/lotteries.json';

// Debug check: does the file exist?
if (!file_exists($jsonFile)) {
    echo "<p style='color:red'>Error: File does not exist at $jsonFile</p>";
    return;
}

// Attempt to read the file
$jsonData = file_get_contents($jsonFile);
if ($jsonData === false) {
    echo "<p style='color:red'>Error: Could not read $jsonFile (check permissions)</p>";
    return;
}

// Decode the JSON
$data = json_decode($jsonData, true);
if (json_last_error() !== JSON_ERROR_NONE) {
    echo "<p style='color:red'>Error decoding JSON: " . json_last_error_msg() . "</p>";
    return;
}

// Access the 'lotteries' array
if (!isset($data['lotteries']) || !is_array($data['lotteries'])) {
    echo "<p style='color:red'>Error: JSON does not have a valid 'lotteries' key.</p>";
    return;
}

$lotteries = $data['lotteries'];

/* -----------------------------------------------------
   1b) SKAI Lab – handle "Save best settings" POST
   ----------------------------------------------------- */
$app   = Factory::getApplication();
$input = $app->getInput();

// Derive a lottery_id for saving settings.
// Prefer gameId / game_id from URL; fall back to generic label.
$lotteryId = (string) $input->getString(
    'gameId',
    $input->getString('game_id', '')
);

// Optional: you can later refine this to map from lotteryIndex → $lotteries[...] if needed.

if ($input->getMethod() === 'POST' && $input->getString('save_skai_best_settings', '') !== '') {
    // CSRF check (Joomla form token)
    if (!Session::checkToken()) {
        $app->enqueueMessage('Invalid session token while saving SKAI settings.', 'error');
    } else {
        $user = Factory::getUser();
        if ($user->guest) {
            $app->enqueueMessage('Please log in to save SKAI settings.', 'warning');
        } else {
            /** @var DatabaseInterface $db */
            $db = Factory::getContainer()->get(DatabaseInterface::class);

            // Setting label from form; fallback if empty
            $settingName = trim((string) $input->getString('setting_name', ''));
            if ($settingName === '') {
                $settingName = 'SKAI Lab – ' . date('Y-m-d');
            }

            // Raw JSON string from hidden field; we will validate it
            $paramsJson = (string) $input->getRaw('params', '');
            $decoded    = json_decode($paramsJson, true);

            if (!is_array($decoded)) {
                $app->enqueueMessage('Could not save SKAI settings: invalid parameter payload.', 'error');
            } else {
                // Re-encode to ensure clean JSON
                $cleanJson = json_encode($decoded, JSON_UNESCAPED_SLASHES | JSON_UNESCAPED_UNICODE);

                try {
                    $query = $db->getQuery(true)
                        ->insert($db->quoteName('#__user_saved_settings'))
                        ->columns([
                            $db->quoteName('user_id'),
                            $db->quoteName('lottery_id'),
                            $db->quoteName('source'),
                            $db->quoteName('setting_name'),
                            $db->quoteName('params'),
                            $db->quoteName('created_at'),
                        ])
                        ->values(implode(',', [
                            (int) $user->id,
                            $db->quote($lotteryId),
                            $db->quote('SKAI_LAB'), // source identifier for this lab
                            $db->quote($settingName),
                            $db->quote($cleanJson),
                            $db->quote(date('Y-m-d H:i:s')),
                        ]));

                    $db->setQuery($query);
                    $db->execute();

                    $app->enqueueMessage('SKAI best settings saved for future use.', 'message');
                } catch (\Throwable $e) {
                    $app->enqueueMessage('Error saving SKAI settings: ' . $e->getMessage(), 'error');
                }
            }
        }
    }
}

/* -----------------------------------------------------
   2) Normalize Lottery State Names & Labels
   ----------------------------------------------------- */
foreach ($lotteries as &$lot) {
    // Normalize state field (handle alternative keys, trim spaces)
    $stateKey = 'state';
    if (!isset($lot[$stateKey])) {
        foreach (['State', 'STATE', ' state', 'state '] as $altKey) {
            if (isset($lot[$altKey])) {
                $stateKey = $altKey;
                break;
            }
        }
    }
    $stateValue = isset($lot[$stateKey]) ? trim($lot[$stateKey]) : '';
    // Force Multi-State for specific game IDs regardless of existing state value
    if (in_array($lot['gameId'], ['101','113','128','132','134','135'])) {
        $stateValue = 'Multi-State';
    } elseif ($stateValue === '') {
        $stateValue = 'Unknown';
    }
    $lot['state'] = $stateValue;
    $lot['label'] = $stateValue . ' - ' . $lot['label'];
}
unset($lot);


// Optionally sort the lotteries by label
usort($lotteries, function($a, $b) {
    return strcmp($a['label'], $b['label']);
});

/* -----------------------------------------------------
   3) Determine Which Lottery Was Chosen by GET param
   ----------------------------------------------------- */
$app   = Factory::getApplication();
$input = $app->getInput();

// Use Joomla input filtering and clamp to valid range
$chosenLotteryIndex = $input->getInt('lotteryIndex', 0);
if ($chosenLotteryIndex < 0 || $chosenLotteryIndex >= count($lotteries)) {
    $chosenLotteryIndex = 0;
}
$chosen = $lotteries[$chosenLotteryIndex];

/* -----------------------------------------------------
   4) Fetch DB Data for the Chosen Lottery
   ----------------------------------------------------- */
$db = Factory::getDbo();
$tableName = $chosen['tableName'];
$gameId    = $chosen['gameId'];

$query = $db->getQuery(true)
    ->select('*')
    ->from($db->quoteName($tableName))
    ->where($db->quoteName('game_id') . ' = ' . $db->quote($gameId))
    ->order($db->quoteName('draw_date') . ' ASC');
$db->setQuery($query);
$historicalResults = $db->loadObjectList();

$draws = [];
$lotteryName = $chosen['label'];
if (!empty($historicalResults) && property_exists($historicalResults[0], 'game_name')) {
    $lotteryName = $historicalResults[0]->game_name;
}
$pickSize = (int) $chosen['pickSize'];
$hasExtra = ((int) $chosen['rangeExtra'] > 0);

foreach ($historicalResults as $row) {
    // Build exactly $pickSize “main” numbers:
    $tempNums = [];
    // Define the sequence of DB fields you might pull
    $fields = ['first','second','third','fourth','fifth','sixth','seventh'];
    for ($i = 0; $i < $pickSize; $i++) {
        $f = $fields[$i] ?? null;
        if ($f && isset($row->$f)) {
            $tempNums[] = (int) $row->$f;
        }
    }
    // If there’s an extra ball, it will be the next field after the mains
    $powerball = null;
    if ($hasExtra) {
        $extraField = $fields[$pickSize] ?? null;
        if ($extraField && isset($row->$extraField)) {
            $powerball = (int) $row->$extraField;
        }
    }
    $draws[] = [
        'numbers'   => $tempNums,
        'powerball' => $powerball,
        'draw_date' => $row->draw_date
    ];
}

$countQuery = $db->getQuery(true)
    ->select('COUNT(*)')
    ->from($db->quoteName($tableName))
    ->where($db->quoteName('game_id') . ' = ' . $db->quote($gameId));
$db->setQuery($countQuery);
$totalEntriesInDB = (int)$db->loadResult();


// === NEW helper to compute “Weighted Hit Score” per draw ===
function weightedHitScore(array $actualMain, ?int $actualBonus,
                          array $predMainTop20, ?int $predictedBonus,
                          int $wMain, int $wBonus): int {
    $mainHits = count(array_intersect($actualMain, $predMainTop20));
    $bonusHit = ($actualBonus !== null && $predictedBonus === $actualBonus) ? 1 : 0;
    return $wMain * $mainHits + $wBonus * $bonusHit;
}

          [[h2]]Setting Explanations[[/h2]]
          [[p]]Expand any section for a plain-language explanation of what that setting does and what values to try.[[/p]]

          [[details]]
            [[summary]][[strong]]Epochs[[/strong]][[/summary]]
            [[p]]One epoch = one full pass of all draw data through the model. More epochs give the model more chances to learn patterns, but too many can cause it to memorise noise (over-fitting). Start with 70–130 and widen the range only if scores plateau. Use a range like [[strong]]70-130[[/strong]] so the lab automatically tests each value.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]Recency Decay[[/strong]][[/summary]]
            [[p]]Controls how much of the draw history is used. [[strong]]1[[/strong]] = use all available draws. [[strong]]0.5[[/strong]] = use only the most recent 50%. Lower values help the model focus on recent trends, but reduce the amount of training data. Try 0.75 and 1 together to see which performs better.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]Skip Strength (Skip Alpha)[[/strong]][[/summary]]
            [[p]]Scales the weight given to skip-pattern features in the neural network's input. [[strong]]0[[/strong]] = the model ignores skips entirely. [[strong]]1[[/strong]] = full weight. Start at 0.5 and explore lower values if skip patterns seem to add noise; raise it if the SKAI engine uses skip heavily in your live settings.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]Main Weight Profiles (Freq / Skip / Hist)[[/strong]][[/summary]]
            [[p]]Sets the relative importance of three signals: [[strong]]Frequency[[/strong]] (how often a number appeared), [[strong]]Skip[[/strong]] (how long since it last appeared), and [[strong]]History[[/strong]] (long-term recurrence). Enter a triplet like [[strong]]40,40,20[[/strong]]. Separate multiple profiles with a pipe (|) to compare them. Values are normalized, so 40/40/20 and 2/2/1 give the same result.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]Scoring Profile[[/strong]][[/summary]]
            [[p]]Determines how the lab ranks configurations after testing. [[strong]]Balanced[[/strong]] rewards a mix of average hits and low volatility. [[strong]]Steady[[/strong]] prioritises consistency (low variance). [[strong]]Aggressive[[/strong]] rewards peak match counts even at the cost of consistency. Choose based on your playing strategy.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]SKAI Analysis Window[[/strong]][[/summary]]
            [[p]]How many of the most recent draws SKAI uses when building its frequency, skip, and history tables. Typical values: 400–800 for weekly games; 300–500 for daily games. A smaller window reacts faster to recent trends; a larger window is more stable over time.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]SKAI Skip Window[[/strong]][[/summary]]
            [[p]]How far back SKAI looks when measuring "skip" – the number of draws since a number last appeared. Separate from the main analysis window. Typical: 30–800. A larger skip window captures more historical skip behaviour.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]Blend – Skip % / AI %[[/strong]][[/summary]]
            [[p]]How much of the final SKAI score comes from skip-pattern signals versus the AI model's probability estimates. At 50 / 50 the two sources are equal. Enter a range in the Skip % field (e.g., [[strong]]10-60[[/strong]]) to have the lab test values 10, 12, 14 … 60 and find the best split for your lottery.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]Sampling – Temperature, Diversity, Gap[[/strong]][[/summary]]
            [[p]]Controls how SKAI generates and diversifies combos. [[strong]]Temperature[[/strong]]: higher values produce more varied picks (0.25–4.0, default 0.8). [[strong]]Diversity Penalty[[/strong]]: reduces the chance of repeating numbers across combos (0–0.5, default 0.1). [[strong]]Gap Scale[[/strong]]: adjusts the spacing between selected numbers (0–1.3, default 1.0).[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]Laplace Smoothing (K)[[/strong]][[/summary]]
            [[p]]Adds a small constant K to all frequency counts before computing probabilities. This prevents numbers that have never appeared from receiving a score of zero. [[strong]]K=1[[/strong]] (default) gives a slight smoothing effect. [[strong]]K=0[[/strong]] uses raw counts. [[strong]]K=2[[/strong]] adds heavier smoothing and is useful when the dataset is small.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]Skip Gamma[[/strong]][[/summary]]
            [[p]]Controls how strongly skip patterns influence SKAI's composite score. The influence follows [[strong]]exp(skipGamma × tanh(z))[[/strong]], so larger values produce a more pronounced skip bias. Default 0.20. Values above 0.5 make SKAI strongly skip-driven; values near 0 effectively disable the skip signal.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]True Exhaustive Mode &amp; Bounded Mode[[/strong]][[/summary]]
            [[p]]When [[strong]]True Exhaustive Mode[[/strong]] is checked, the lab tests every possible combination of all settings. The combination count shown above the run button reflects the total. For large searches this can be very slow — the live count updates as you change inputs. [[strong]]Bounded Mode[[/strong]] (default, unchecked) caps the search at 500 combinations for a quick result; use it to validate a setup before a full run.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]Performance Metrics (F1, Avg Matches, Composite Score)[[/strong]][[/summary]]
            [[p]]After each run the lab displays: [[strong]]Avg Matches[[/strong]] = average correct numbers per validation draw; [[strong]]F1 Score[[/strong]] = balance between precision and recall across all numbers; [[strong]]F1 Std Dev[[/strong]] = how consistent the F1 is across validation windows (lower is more stable); [[strong]]Composite Score[[/strong]] = the final ranking metric that combines all of the above according to the chosen Scoring Profile.[[/p]]
          [[/details]]

          [[details]]
            [[summary]][[strong]]Smart Narrowing – Pool, Rounds, Keep %, Seed[[/strong]][[/summary]]
            [[p]]Smart Narrowing uses a tournament-style search ([[strong]]successive halving[[/strong]]) to find the best settings without testing every combination in full. It starts by sampling a random set of configurations (the [[strong]]Pool[[/strong]]), trains each one briefly, then at the end of every [[strong]]Round[[/strong]] keeps only the top [[strong]]Keep %[[/strong]] performers and discards the rest. The survivors train progressively longer each round until only the very best remain.[[/p]]
            [[p]][[strong]]Pool[[/strong]] — the total number of random configurations sampled at the start. A larger pool explores more of the search space but takes longer. [[strong]]Good starting range: 80–200.[[/strong]] Increase to 300+ only if you suspect the best settings are in a narrow, hard-to-find region.[[/p]]
            [[p]][[strong]]Rounds[[/strong]] — how many elimination rounds to run. Each round trains survivors longer and removes the weakest performers. [[strong]]3–5 rounds[[/strong]] is the practical sweet-spot. Fewer rounds finish faster but may not converge; more than 6 rarely helps unless the pool is very large.[[/p]]
            [[p]][[strong]]Keep %[[/strong]] — what fraction of the current pool advances after each round. At [[strong]]25 %[[/strong]] (default), 75 % of configs are cut each round, so the field narrows quickly. Lower values (15–20 %) converge faster but risk losing good configs early. Higher values (35–40 %) are safer but slower.[[/p]]
            [[p]][[strong]]Seed[[/strong]] — the random seed for sampling the initial pool. [[strong]]0 = auto[[/strong]] (a fresh seed is drawn from the timestamp every run — recommended for exploration). Enter a fixed number to reproduce a previous run exactly, which is useful for comparing changes to other settings.[[/p]]
            [[p]][[strong]]Recommended starting point:[[/strong]] Pool 120 · Rounds 4 · Keep 25 % · Seed 0. If top scores vary widely between runs, increase the Pool. If results look good but you want to verify reproducibility, re-run with the same fixed Seed.[[/p]]
          [[/details]]

function _padStart(str, len, ch) {
  str = String(str);
  if (!ch) { ch = ' '; }
  while (str.length < len) { str = ch + str; }
  return str;
}

// ---- helper: compute a deterministic run signature from settings ----
function computeRunSignature(settingsSnapshot) {
  var keys = Object.keys(settingsSnapshot).sort();
  var str  = keys.map(function(k){ return k + '=' + settingsSnapshot[k]; }).join('|');
  var h = 0;
  for (var i = 0; i < str.length; i++) {
    h = ((h << 5) - h + str.charCodeAt(i)) | 0;
  }
  return 'sig-' + _padStart(Math.abs(h).toString(16).toUpperCase(), 8, '0');
}

function computeSensitivityReport(results) {
  // results: array of topSettings objects (each has all setting fields + compositeScore)
  if (!results || results.length < 2) { return []; }

  var settings = [
    { key:'epochs',           label:'Epochs' },
    { key:'recencyDecay',     label:'Recency Decay' },
    { key:'skipAlpha',        label:'Skip Alpha' },
    { key:'skaiBlendSkipPct', label:'Blend Skip %' },
    { key:'skaiSamplingTemp', label:'Temperature' },
    { key:'skaiDiversityPen', label:'Diversity Pen' },
    { key:'skaiGapScale',     label:'Gap Scale' },
    { key:'laplaceK',         label:'Laplace K' },
    { key:'skipGamma',        label:'Skip Gamma' }
  ];

  var report = [];
  for (var si = 0; si < settings.length; si++) {
    var sd = settings[si];
    var key = sd.key;

    // Group results by distinct value of this setting
    var groups = {};
    for (var ri = 0; ri < results.length; ri++) {
      var r = results[ri];
      var val = r[key];
      if (val === undefined || val === null) { continue; }
      var vStr = String(val);
      if (!groups[vStr]) { groups[vStr] = []; }
      groups[vStr].push(r.compositeScore);
    }

    var groupKeys = Object.keys(groups);
    if (groupKeys.length < 2) { continue; } // not enough variation to measure impact

    // Compute mean score per value group
    var groupMeans = {};
    for (var g = 0; g < groupKeys.length; g++) {
      var gk = groupKeys[g];
      var arr = groups[gk];
      var sum = 0;
      for (var ai = 0; ai < arr.length; ai++) { sum += arr[ai]; }
      groupMeans[gk] = sum / arr.length;
    }

    // Overall mean score
    var allScores = results.map(function(r){ return r.compositeScore; });
    var overallMean = allScores.reduce(function(a,b){ return a+b; }, 0) / allScores.length;

    // Impact = max deviation of group mean from overall mean
    var maxImpact = 0;
    var bestVal = '';
    var direction = 0;
    for (var g = 0; g < groupKeys.length; g++) {
      var gk = groupKeys[g];
      var dev = groupMeans[gk] - overallMean;
      if (Math.abs(dev) > Math.abs(maxImpact)) {
        maxImpact = dev;
        bestVal = gk;
      }
    }
    direction = maxImpact > 0 ? '+' : (maxImpact < 0 ? '−' : '0');

    // Variance across group means
    var allMeans = groupKeys.map(function(gk){ return groupMeans[gk]; });
    var meanOfMeans = allMeans.reduce(function(a,b){ return a+b; }, 0) / allMeans.length;
    var variance = allMeans.reduce(function(acc, m){ return acc + Math.pow(m - meanOfMeans, 2); }, 0) / allMeans.length;

    report.push({
      label:       sd.label,
      valuesTested: groupKeys.length,
      impact:      maxImpact,
      variance:    variance,
      direction:   direction,
      bestVal:     bestVal
    });
  }

  // Sort by absolute impact descending
  report.sort(function(a, b){ return Math.abs(b.impact) - Math.abs(a.impact); });
  return report;
}


// 1) Table Sort
function sortTable(tableId, columnIndex){
  var table = document.getElementById(tableId);
  if(!table)return;
  var tbody = table.querySelector("tbody");
  var rows = Array.prototype.slice.call(tbody.querySelectorAll("tr"));
  var asc = (table.getAttribute("data-sortcol") == columnIndex && table.getAttribute("data-sortdir")=="asc") ? false : true;
  rows.sort(function(a, b) {
    var aText = a.cells[columnIndex].innerText.trim();
    var bText = b.cells[columnIndex].innerText.trim();
    var aNum = parseFloat(aText.replace(/[^\d.-]/g,''));
    var bNum = parseFloat(bText.replace(/[^\d.-]/g,''));
    if(!isNaN(aNum) && !isNaN(bNum))
      return asc ? aNum - bNum : bNum - aNum;
    else
      return asc ? aText.localeCompare(bText) : bText.localeCompare(aText);
  });
  rows.forEach(function(r) { tbody.removeChild(r); });
  rows.forEach(function(r) { tbody.appendChild(r); });
  table.setAttribute("data-sortcol", columnIndex);
  table.setAttribute("data-sortdir", asc ? "asc" : "desc");
}

// 2) customBinaryCrossentropy + reduceLROnPlateau
function customBinaryCrossentropy(yTrue, yPred){
  var EPS = 1e-7;
  return tf.tidy(function(){
    var clipped = yPred.clipByValue(EPS, 1-EPS);
    var term1 = yTrue.mul(clipped.log()).neg();
    var term2 = tf.scalar(1).sub(yTrue).mul(tf.scalar(1).sub(clipped).log()).neg();
    return term1.add(term2).mean();
  });
}
function reduceLROnPlateau(opt, factor, patience, minLr){
  if (factor === undefined) { factor = 0.5; }
  if (patience === undefined) { patience = 4; }
  if (minLr === undefined) { minLr = 1e-7; }
  var bestVal = Infinity, wait = 0;
  return {
    onEpochEnd: async function(ep, logs) {
      var valLoss = logs.val_loss;
      if(valLoss < bestVal){
        bestVal = valLoss; wait = 0;
      } else {
        wait++;
        if(wait >= patience){
          var lr;
          if(typeof opt.learningRate === 'object'){
            lr = opt.learningRate.dataSync()[0];
          } else {
            lr = opt.learningRate;
          }
          var newLr = lr * factor; if(newLr < minLr)newLr = minLr;
          if(typeof opt.learningRate === 'object'){
            opt.learningRate.assign(newLr);
          } else {
            opt.learningRate = newLr;
          }
          console.log('ReduceLROnPlateau => epoch ' + ep + ', new lr: ' + newLr);
          wait = 0;
        }
      }
    }
  };
}

// 3) For epoch-end distribution updates
function computeEpochDist(predArr, valArr, N){
  var topK = 20;
  var dist = [0,0,0,0,0,0];
  for(var i = 0; i < valArr.length; i++){
    var mainProbs = predArr[i].slice(0, N);
    var sorted = mainProbs.map(function(v, idx) { return { v: v, index: idx+1 }; });
    sorted.sort(function(a, b) { return b.v - a.v; });
    var topIdx = sorted.slice(0, topK).map(function(o) { return o.index; });
    var actual = [];
    for(var j = 0; j < N; j++){
      if(valArr[i][j] === 1) actual.push(j+1);
    }
    var c = actual.filter(function(num) { return topIdx.indexOf(num) !== -1; }).length;
    if(c >= 0 && c < dist.length){
      dist[c]++;
    }
  }
  return dist;
}

  function computeSkipFeatureSeries(drawsNumbers, pickSize, mainNumbersMax) {
    var lastHitIndex = new Array(mainNumbersMax).fill(null);
    var skipsPerStep = [];

    for (var t = 0; t < drawsNumbers.length; t++) {
      var skipVec = new Array(mainNumbersMax).fill(0);

      // For each number, how long since it last appeared?
      for(var k = 0; k < mainNumbersMax; k++) {
        var last = lastHitIndex[k];
        var skip = (last === null) ? t : (t - last - 1);
        skipVec[k] = skip;
      }

      // Update lastHitIndex using the CURRENT draw's main numbers.
      var draw = drawsNumbers[t];
      for(var j = 0; j < pickSize; j++) {
        var n = draw[j];
        if (n >= 1 && n <= mainNumbersMax) {
          lastHitIndex[n - 1] = t;
        }
      }

      skipsPerStep.push(skipVec);
    }

    // Normalize skips to 0..1 so they combine safely with other features.
    var maxSkip = 0;
    for(var t = 0; t < skipsPerStep.length; t++) {
      var vec = skipsPerStep[t];
      for(var i = 0; i < vec.length; i++) {
        if (vec[i] > maxSkip) {
          maxSkip = vec[i];
        }
      }
    }

    if (maxSkip > 0) {
      for(var t = 0; t < skipsPerStep.length; t++) {
        var vec = skipsPerStep[t];
        for(var i = 0; i < vec.length; i++) {
          vec[i] = vec[i] / maxSkip;
        }
      }
    }

    return skipsPerStep;

  function evaluateThresholds(predArr, valArr, thresholds){
    var bestF1 = -Infinity, bestThr = 0.5;
    for(var _ti = 0; _ti < thresholds.length; _ti++){
      var thr = thresholds[_ti];
      var tp = 0, fp = 0, fn = 0;
      for(var i = 0; i < valArr.length; i++){
        for(var j = 0; j < N; j++){
          var predicted = predArr[i][j] > thr ? 1 : 0;
          var actual = valArr[i][j];
          if(predicted === 1 && actual === 1) tp++;
          else if(predicted === 1 && actual === 0) fp++;
          else if(predicted === 0 && actual === 1) fn++;
        }
      }
      var prec = (tp + fp) === 0 ? 0 : tp / (tp + fp);
      var rec = (tp + fn) === 0 ? 0 : tp / (tp + fn);
      var f1 = (prec + rec) === 0 ? 0 : 2 * (prec * rec) / (prec + rec);
      if(f1 > bestF1){
        bestF1 = f1; bestThr = thr;
      }
    }
    return { bestF1: bestF1, bestThr: bestThr };
  }

function skaiRestoreLastSettings() {
  try {
    var raw = localStorage.getItem(LS_LAST_SETTINGS_KEY);
    if (!raw) { return; }
    var snap = JSON.parse(raw);
    if (!snap) { return; }

    // Restore sliders
    if (snap.sliders) {
      Object.keys(snap.sliders).forEach(function(id) {
        var s = snap.sliders[id];
        if (s && s.lo != null && typeof setSliderRange === 'function') {
          setSliderRange(id, parseFloat(s.lo), parseFloat(s.hi != null ? s.hi : s.lo));
        }
      });
    }

    // Restore plain inputs (select, number)
    if (snap.inputs) {
      Object.keys(snap.inputs).forEach(function(id) {
        var el = document.getElementById(id);
        if (el) { el.value = snap.inputs[id]; }
      });
    }

    // Restore checkboxes and fire change events so dependent UI updates
    if (snap.checks) {
      Object.keys(snap.checks).forEach(function(id) {
        var el = document.getElementById(id);
        if (!el) { return; }
        el.checked = snap.checks[id];
        el.dispatchEvent(new Event('change', { bubbles: true }));
      });
    }

    // Refresh the combos count display after restoration
    if (typeof updateCombosCount === 'function') {
      try { updateCombosCount(); } catch(e) { /* non-critical */ }
    }
  } catch(e) { /* ignore parse / restore errors */ }
}

function applyPreset(values) {
  // Update hidden inputs (the slider JS reads these on sync)
  if (values.epochs !== undefined)  { document.getElementById('epochsRange').value       = values.epochs; }
  if (values.recency !== undefined) { document.getElementById('recencyDecayRates').value = values.recency; }

  // Sync range sliders from the new hidden-input values
  function parseMinMax(str) {
    str = String(str || '').trim();
    if (str.indexOf(',') !== -1) {
      var nums = str.split(',').map(parseFloat).filter(function(v){ return !isNaN(v); });
      if (!nums.length) { return [NaN, NaN]; }
      return [Math.min.apply(null, nums), Math.max.apply(null, nums)];
    }
    if (str.indexOf('-') !== -1) {
      var pts = str.split('-');
      return [parseFloat(pts[0]), parseFloat(pts[1] || pts[0])];
    }
    var s = parseFloat(str);
    return [s, s];
  }
  if (values.epochs !== undefined)  { var ep = parseMinMax(values.epochs);  if (typeof setSliderRange === 'function') { setSliderRange('epochsRange',       ep[0], ep[1]); } }
  if (values.recency !== undefined) { var rc = parseMinMax(values.recency); if (typeof setSliderRange === 'function') { setSliderRange('recencyDecayRates', rc[0], rc[1]); } }
  if (values.skip !== undefined)    { var sk = parseMinMax(values.skip);    if (typeof setSliderRange === 'function') { setSliderRange('skipAlphaValues',   sk[0], sk[1]); } }
}

  function makePRNG(seed) {
    var s = seed >>> 0;
    return function () {
      s += 0x6D2B79F5;
      var t = Math.imul(s ^ (s >>> 15), 1 | s);
      t = t + Math.imul(t ^ (t >>> 7), 61 | t) ^ t;
      return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
    };
  }

  function snmPickFrom(rng, arr) {
    return arr[Math.floor(rng() * arr.length)];
  }

  function snmGenerateCandidates(poolSize, rng) {
    var epochOptions   = [70, 100, 130];
    var decayOptions   = [0, 0.25, 0.5, 0.75, 1];
    var alphaOptions   = [0, 0.25, 0.5, 0.75, 1];
    var actOptions     = ['tanh', 'relu', 'sigmoid'];
    var layersOptions  = [1, 2, 3];
    var laplaceOptions = [0, 1, 2];
    var gammaOptions   = [0, 0.1, 0.2, 0.3, 0.5];
    var blendOptions   = [30, 50, 70, 100];

    var seen = {};
    var candidates = [];
    var maxTries = poolSize * 20;

    for (var i = 0; candidates.length < poolSize && i < maxTries; i++) {
      var cfg = {
        epochs:            snmPickFrom(rng, epochOptions),
        recencyDecay:      snmPickFrom(rng, decayOptions),
        skipAlpha:         snmPickFrom(rng, alphaOptions),
        activationFunction: snmPickFrom(rng, actOptions),
        numHiddenLayers:   snmPickFrom(rng, layersOptions),
        laplaceK:          snmPickFrom(rng, laplaceOptions),
        skipGamma:         snmPickFrom(rng, gammaOptions),
        blendSkipPct:      snmPickFrom(rng, blendOptions),
        // Defaults from SKAI_SETTINGS_REGISTRY
        batchSize:         SKAI_SETTINGS_REGISTRY.batchSize.defaultVal,
        dropoutRate:       SKAI_SETTINGS_REGISTRY.dropoutRate.defaultVal,
        learningRate:      SKAI_SETTINGS_REGISTRY.learningRate.defaultVal
      };
      var sig = snmConfigSig(cfg);
      if (!seen[sig]) {
        seen[sig] = true;
        candidates.push(cfg);
      }
    }
    return candidates;
  }

  function snmConfigSig(cfg) {
    var keys = ['epochs','recencyDecay','skipAlpha','activationFunction','numHiddenLayers',
                'laplaceK','skipGamma','blendSkipPct'];
    return keys.map(function(k){ return k + ':' + cfg[k]; }).join('|');
  }

  function snmNeighbors(cfg) {
    var neighbors = [];
    var epochOpts  = [70, 100, 130];
    var decayOpts  = [0, 0.25, 0.5, 0.75, 1];
    var alphaOpts  = [0, 0.25, 0.5, 0.75, 1];

    function stepAround(arr, val) {
      var idx = arr.indexOf(val);
      var result = [];
      if (idx > 0)               { result.push(arr[idx - 1]); }
      if (idx < arr.length - 1)  { result.push(arr[idx + 1]); }
      return result;
    }

    var eNeighbors = stepAround(epochOpts, cfg.epochs);
    var dNeighbors = stepAround(decayOpts, cfg.recencyDecay);
    var aNeighbors = stepAround(alphaOpts, cfg.skipAlpha);

    for (var ei = 0; ei < eNeighbors.length; ei++) {
      var n = {};
      for (var k in cfg) { if (cfg.hasOwnProperty(k)) { n[k] = cfg[k]; } }
      n.epochs = eNeighbors[ei];
      neighbors.push(n);
    }
    for (var di = 0; di < dNeighbors.length; di++) {
      var n2 = {};
      for (var k2 in cfg) { if (cfg.hasOwnProperty(k2)) { n2[k2] = cfg[k2]; } }
      n2.recencyDecay = dNeighbors[di];
      neighbors.push(n2);
    }
    for (var ai = 0; ai < aNeighbors.length; ai++) {
      var n3 = {};
      for (var k3 in cfg) { if (cfg.hasOwnProperty(k3)) { n3[k3] = cfg[k3]; } }
      n3.skipAlpha = aNeighbors[ai];
      neighbors.push(n3);
    }
    return neighbors;
  }

  async function snmEvaluateConfig(cfg, epochBudget) {
    // Evaluate one config against globalData (already prepared).
    // Returns { compositeScore, avgMatches, f1Score, signature }
    var signature = snmConfigSig(cfg);
    if (!globalData) { return { compositeScore: 0, avgMatches: 0, f1Score: 0, signature: signature }; }

    var epochs           = Math.max(5, Math.min(1000, epochBudget || cfg.epochs));
    var batchSize        = cfg.batchSize     || 4;
    var dropoutRate      = cfg.dropoutRate   || 0.2;
    var learningRate     = cfg.learningRate  || 0.002;
    var activationFunction = cfg.activationFunction || 'tanh';
    var numHiddenLayers  = cfg.numHiddenLayers || 2;
    var recencyDecay     = (cfg.recencyDecay != null) ? cfg.recencyDecay : 1.0;
    var skipAlpha        = (cfg.skipAlpha    != null) ? cfg.skipAlpha    : 0.5;
    var laplaceK         = (cfg.laplaceK     != null) ? cfg.laplaceK     : 1;
    var skipGamma        = (cfg.skipGamma    != null) ? cfg.skipGamma    : 0.2;

    var inputs       = globalData.inputs;
    var labels       = globalData.labels;
    var numExamples  = globalData.numExamples;
    var inputMin     = globalData.inputMin;
    var inputMax     = globalData.inputMax;
    var inputDim     = pickSize + N;

    // Scale skip features by skipAlpha
    var scaleArr = [];
    for (var i = 0; i < pickSize; i++) { scaleArr.push(1); }
    for (var j = 0; j < N; j++)        { scaleArr.push(skipAlpha); }
    var featureScale  = tf.tensor1d(scaleArr);
    var inputsScaled  = inputs.mul(featureScale);

    // Recency subset
    var rdClamped     = Math.max(0, Math.min(1, recencyDecay));
    var fracHist      = 0.3 + (1.0 - 0.3) * rdClamped;
    var localNumEx    = Math.floor(numExamples * fracHist);
    if (localNumEx < 9) { localNumEx = Math.min(numExamples, 9); }
    var effStart      = numExamples - localNumEx;
    var localInputs   = inputsScaled.slice([effStart, 0], [localNumEx, -1]);
    var localLabels   = labels.slice([effStart, 0], [localNumEx, -1]);

    // Use 1 rolling window for speed (early rounds) or 3 for late rounds
    var numWin        = epochBudget >= 100 ? 3 : 1;
    var localWinSize  = Math.max(1, Math.floor(localNumEx / (numWin * 3)));

    var totalF1 = 0, matchCount = 0, totalValEx = 0;
    var foldF1s = [];
    var perDrawWHS = [];
    var drawsWith3Plus = 0, drawsWith4Plus = 0, drawsWithBonusHit = 0;
    var snmFoldLosses = [];    // per-epoch training losses across all folds
    var snmFoldValLosses = []; // per-epoch validation losses across all folds
    var snmMatchDist  = [0,0,0,0,0,0]; // match distribution for this candidate
    var wMain  = <?php echo (int)($chosen['mainWeight'] ?? 5); ?>;
    var wBonus = <?php echo (int)($chosen['bonusWeight'] ?? 2); ?>;

    for (var w = 0; w < numWin; w++) {
      var trainEnd = (w + 2) * localWinSize;
      if (trainEnd >= localNumEx) { break; }
      var valStart = trainEnd;
      var valEnd   = Math.min(trainEnd + localWinSize, localNumEx);
      var valCount = valEnd - valStart;
      if (valCount <= 0) { break; }

      var trainX = localInputs.slice([0, 0], [trainEnd, -1]);
      var trainY = localLabels.slice([0, 0], [trainEnd, -1]);
      var valX   = localInputs.slice([valStart, 0], [valCount, -1]);
      var valY   = localLabels.slice([valStart, 0], [valCount, -1]);

      var model = tf.sequential();
      model.add(tf.layers.dense({ units: 128, activation: activationFunction, inputShape: [inputDim] }));
      model.add(tf.layers.dropout({ rate: dropoutRate }));
      for (var li = 0; li < numHiddenLayers; li++) {
        model.add(tf.layers.dense({ units: 64, activation: activationFunction }));
        model.add(tf.layers.dropout({ rate: dropoutRate }));
      }
      model.add(tf.layers.dense({ units: N + M, activation: 'sigmoid' }));
      var opt = tf.train.adam(learningRate);
      model.compile({ optimizer: opt, loss: customBinaryCrossentropy, metrics: ['binaryAccuracy'] });

      await model.fit(trainX, trainY, {
        epochs: epochs,
        batchSize: batchSize,
        validationData: [valX, valY],
        verbose: 0,
        callbacks: {
          onEpochEnd: async function(ep, logs) {
            if (logs && logs.loss != null)     { snmFoldLosses.push(logs.loss); }
            if (logs && logs.val_loss != null) { snmFoldValLosses.push(logs.val_loss); }
            // Update live status ticker and epoch progress bar for SNM candidates
            updateLiveStatusTicker(ep + 1, epochs, logs.loss, logs.val_loss, 'SNM \u2013 ');
            if (epC)  { epC.style.display = 'block'; }
            if (epBar){ epBar.style.width = Math.round(((ep + 1) / epochs) * 100) + '%'; }
            await tf.nextFrame();
          }
        }
      });

      var preds    = model.predict(valX);
      var predArr  = await preds.array();
      var valArr   = await valY.array();

      // accumulate match distribution for this candidate
      var foldDist = computeEpochDist(predArr, valArr, N);
      for (var di = 0; di < snmMatchDist.length; di++) { snmMatchDist[di] += foldDist[di]; }

      var tp = 0, fp = 0, fn = 0;
      var foldMatchCount = 0;
      totalValEx += valCount;

      for (var vi = 0; vi < valArr.length; vi++) {
        var mainProbs  = predArr[vi].slice(0, N);
        var extraProbs = predArr[vi].slice(N, N + M);

        var sortedMain = mainProbs.map(function(v, idx){ return { v: v, index: idx + 1 }; });
        sortedMain.sort(function(a, b){ return b.v - a.v; });
        var topIdx = sortedMain.slice(0, 20).map(function(o){ return o.index; });

        var actualArray = [];
        for (var ai2 = 0; ai2 < N; ai2++) {
          if (valArr[vi][ai2] === 1) { actualArray.push(ai2 + 1); }
        }

        var matchedArr2 = actualArray.filter(function(n){ return topIdx.indexOf(n) !== -1; });
        foldMatchCount += matchedArr2.length;
        var hitsThisDraw = matchedArr2.length;
        if (hitsThisDraw >= 3) { drawsWith3Plus++; }
        if (hitsThisDraw >= 4) { drawsWith4Plus++; }

        var actualEB2 = null;
        for (var ex2 = 0; ex2 < M; ex2++) {
          if (valArr[vi][N + ex2] === 1) { actualEB2 = ex2 + 1; }
        }
        var eBest2 = -1, eVal2 = -Infinity;
        for (var ex3 = 0; ex3 < M; ex3++) {
          if (extraProbs[ex3] > eVal2) { eVal2 = extraProbs[ex3]; eBest2 = ex3 + 1; }
        }
        var bonusHit2 = (actualEB2 && eBest2 === actualEB2) ? 1 : 0;
        var whs2 = wMain * matchedArr2.length + wBonus * bonusHit2;
        perDrawWHS.push(whs2);
        if (bonusHit2) { drawsWithBonusHit++; }

        for (var fi2 = 0; fi2 < N; fi2++) {
          var predicted2 = predArr[vi][fi2] > 0.5 ? 1 : 0;
          var act2       = valArr[vi][fi2];
          if (predicted2 === 1 && act2 === 1)      { tp++; }
          else if (predicted2 === 1 && act2 === 0) { fp++; }
          else if (predicted2 === 0 && act2 === 1) { fn++; }
        }
      }

      var prec2  = (tp + fp) === 0 ? 0 : tp / (tp + fp);
      var rec2   = (tp + fn) === 0 ? 0 : tp / (tp + fn);
      var foldF1 = (prec2 + rec2 === 0) ? 0 : 2 * (prec2 * rec2) / (prec2 + rec2);
      foldF1s.push(foldF1);
      totalF1     += foldF1;
      matchCount  += foldMatchCount;

      preds.dispose();
      trainX.dispose(); trainY.dispose();
      valX.dispose(); valY.dispose();
      model.dispose();
    }

    featureScale.dispose();
    inputsScaled.dispose();
    localInputs.dispose();
    localLabels.dispose();

    var usedWin   = foldF1s.length || 1;
    var avgF1     = totalF1 / usedWin;
    var denomD    = totalValEx || 1;
    var avgMatches = matchCount / denomD;
    var muWHS2    = perDrawWHS.length
      ? perDrawWHS.reduce(function(sum, x){ return sum + x; }, 0) / perDrawWHS.length
      : 0;
    var sigmaWHS2 = perDrawWHS.length
      ? Math.sqrt(perDrawWHS.reduce(function(acc, x){ return acc + Math.pow(x - muWHS2, 2); }, 0) / perDrawWHS.length)
      : 0;
    var p3b = drawsWith3Plus / denomD;
    var p4b = drawsWith4Plus / denomD;
    var pBonus2 = drawsWithBonusHit / denomD;

    var varF12 = foldF1s.length
      ? foldF1s.reduce(function(acc, v){ return acc + Math.pow(v - avgF1, 2); }, 0) / foldF1s.length
      : 0;
    var stdF12 = Math.sqrt(varF12);

    // Balanced profile scoring weights
    var SNM_W_MEAN_WHS    = 0.40;  // weight for mean weighted hit score
    var SNM_W_DRAW_PROB   = 0.30;  // weight for tail draw probability (p3+, p4+)
    var SNM_W_STABILITY   = 0.30;  // weight for fold stability (1/stdF1)
    var SNM_W_VARIANCE    = 0.40;  // penalty weight for WHS variance
    var SNM_W_BONUS       = 0.75;  // bonus ball reward weight
    var compositeScore = (SNM_W_MEAN_WHS * muWHS2 + SNM_W_DRAW_PROB * ((p3b + p4b) / 2) + SNM_W_STABILITY * (1 / (1 + stdF12)))
                         - SNM_W_VARIANCE * sigmaWHS2
                         + SNM_W_BONUS * pBonus2;

    return {
      compositeScore: compositeScore,
      avgMatches: avgMatches,
      f1Score: avgF1,
      signature: signature,
      cfg: cfg,
      trainingLosses: snmFoldLosses.slice(),
      validationLosses: snmFoldValLosses.slice(),
      matchDist: snmMatchDist.slice()
    };
  }

  // Helper: estimate total candidate evaluations for a Smart Narrowing run before it starts.
  // Uses: round 1 = poolSize; each subsequent round = max(2, floor(prev * keepPct/100)).
  function snmEstimateTotalEvals(poolSize, numRounds, keepPct) {
    var total = 0;
    var currentPool = poolSize;
    for (var r = 0; r < numRounds; r++) {
      total += currentPool;
      currentPool = Math.max(2, Math.floor(currentPool * keepPct / 100));
    }
    return total;
  }

  // Smart Narrowing algorithm constants
  var SNM_WILDCARD_RATIO          = 0.10; // fraction of pool size to add as wildcards each round
  var SNM_DIVERSITY_SWAP_THRESHOLD = 0.80; // diversity swaps only if candidate scores above this percentile of promoted
  var SNM_LOCAL_REFINEMENT_START_ROUND = 2; // 0-indexed: start local neighborhood refinement from round 3
  var SNM_WILDCARD_GEN_MULTIPLIER = 5;    // over-sample by this factor to find unique wildcard configs

  async function runSmartNarrowing() {
    var snmPoolSizeEl  = document.getElementById('snmPoolSize');
    var snmRoundsEl    = document.getElementById('snmRounds');
    var snmKeepPctEl   = document.getElementById('snmKeepPct');
    var snmSeedEl      = document.getElementById('snmSeed');
    var snmLogEl       = document.getElementById('snmRoundsLog');
    var snmPanel       = document.getElementById('snmRoundsPanel');

    var poolSize  = snmPoolSizeEl  ? Math.max(20, parseInt(snmPoolSizeEl.value,  10) || 120) : 120;
    var numRounds = snmRoundsEl    ? Math.max(2,  parseInt(snmRoundsEl.value,    10) || 4)   : 4;
    var keepPct   = snmKeepPctEl   ? Math.max(10, parseInt(snmKeepPctEl.value,   10) || 25)  : 25;
    var seedVal   = snmSeedEl      ? parseInt(snmSeedEl.value, 10) : 0;
    if (seedVal === 0 || isNaN(seedVal)) { seedVal = (Date.now() & 0xFFFFFF); }

    // Estimate total evaluations and initialise global progress UI before any round starts.
    totalPossible = snmEstimateTotalEvals(poolSize, numRounds, keepPct);
    totalPossibleAnalysesSpan.innerText = totalPossible;
    totalAnalyses = 0;
    totalAnalysesSpan.innerText = 0;
    var snmCombosDisp = document.getElementById('combosCountDisplay');
    if (snmCombosDisp) { snmCombosDisp.textContent = totalPossible.toLocaleString(); }
    loadingMsg.innerText = 'Smart Narrowing is evaluating candidate settings. Live charts will refresh when a candidate completes.';

    // Stable global timer and counter for the entire Smart Narrowing run.
    var smartStartTime = Date.now();
    var smartDone = 0;

    var rng = makePRNG(seedVal);
    var epochBudgets = [25, 50, 100];
    // Fill remaining rounds with the final UI epoch value or 130
    var epochsEl = document.getElementById('epochsRange');
    var maxEpochs = epochsEl ? (parseInt(epochsEl.value, 10) || 130) : 130;
    while (epochBudgets.length < numRounds) { epochBudgets.push(maxEpochs); }
    epochBudgets = epochBudgets.slice(0, numRounds);

    if (snmLogEl)  { snmLogEl.innerHTML = ''; }
    if (snmPanel)  { snmPanel.style.display = 'block'; }

    // Clear stale chart data so charts don't appear frozen from a previous run.
    trainingLosses = [];
    validationLosses = [];
    updateLossChart();
    matchDistribution = [0, 0, 0, 0, 0, 0];
    updateMatchDistChart();

    // Reset best-settings panel so stale results from a previous run don't show.
    topSettings = [];
    var tsBodyReset = document.getElementById('topSettingsBody');
    if (tsBodyReset) { tsBodyReset.innerHTML = ''; }
    var tsElReset = document.getElementById('topSettings');
    if (tsElReset) { tsElReset.style.display = 'none'; }
    var top3Reset = document.getElementById('skaiTop3Cards');
    if (top3Reset) { top3Reset.style.display = 'none'; }
    var bestSumReset = document.getElementById('bestSettingsSummary');
    if (bestSumReset) { bestSumReset.style.display = 'none'; }
    var bestLblReset = document.getElementById('currentBestRunLabel');
    if (bestLblReset) { bestLblReset.style.display = 'none'; bestLblReset.innerText = ''; }

    // Track best-so-far composite score for live chart updates.
    var snmBestScore = -Infinity;

    // Generate initial candidate pool
    var allCandidates = snmGenerateCandidates(poolSize, rng);
    var survivors = allCandidates.slice(); // start with full pool
    var seenSigs  = {};
    for (var ci = 0; ci < survivors.length; ci++) {
      seenSigs[snmConfigSig(survivors[ci])] = true;
    }

    var roundResults = [];

    for (var round = 0; round < numRounds; round++) {
      var budget = epochBudgets[round];
      var roundScores = [];
      var poolLen = survivors.length;

      for (var ci2 = 0; ci2 < survivors.length; ci2++) {
        if (isPaused) {
          await new Promise(function(res){
            var iv = setInterval(function(){ if(!isPaused){ clearInterval(iv); res(); } }, 200);
          });
        }
        // Show "Training…" status before evaluation begins so users see activity
        pbText.innerText = 'Smart Narrowing \u2013 Round ' + (round + 1) + '/' + numRounds +
                           ' | Training candidate ' + (ci2 + 1) + '/' + poolLen +
                           ' | Overall ' + smartDone + '/' + totalPossible;
        if (epText) {
          epText.innerText = 'Round ' + (round + 1) + '/' + numRounds +
            ' \u2013 Candidate ' + (ci2 + 1) + '/' + poolLen + ' \u2013 training\u2026';
        }
        var scored = await snmEvaluateConfig(survivors[ci2], budget);
        smartDone++;
        totalAnalyses = smartDone;
        totalAnalysesSpan.innerText = smartDone;
        pbText.innerText = 'Smart Narrowing \u2013 Round ' + (round + 1) + '/' + numRounds +
                           ' | Candidate ' + (ci2 + 1) + '/' + poolLen +
                           ' | Overall ' + smartDone + '/' + totalPossible;
        updateProgress(smartDone, totalPossible, smartStartTime);
        // Update charts and best settings when a new best candidate is found.
        if (scored.compositeScore > snmBestScore) {
          snmBestScore = scored.compositeScore;
          if (scored.trainingLosses && scored.trainingLosses.length) {
            trainingLosses = scored.trainingLosses.slice();
            validationLosses = (scored.validationLosses && scored.validationLosses.length)
              ? scored.validationLosses.slice() : [];
            updateLossChart();
          }
          if (scored.matchDist) {
            matchDistribution = scored.matchDist.slice();
            updateMatchDistChart();
          }
          snmShowInterimBest(scored);
        }
        roundScores.push(scored);
        await tf.nextFrame();
      }

      // Sort by compositeScore descending
      roundScores.sort(function(a, b){ return b.compositeScore - a.compositeScore; });

      var keepCount   = Math.max(2, Math.floor(poolLen * keepPct / 100));
      var wildCount   = Math.max(1, Math.floor(poolSize * SNM_WILDCARD_RATIO));
      var promoted    = roundScores.slice(0, keepCount);
      var eliminated  = roundScores.slice(keepCount);

      // Diversity enforcement: ensure at least one of each activation + layer count in top
      var actCounts  = {};
      var layerCounts = {};
      for (var pi = 0; pi < promoted.length; pi++) {
        var act = promoted[pi].cfg.activationFunction;
        var lay = promoted[pi].cfg.numHiddenLayers;
        actCounts[act]   = (actCounts[act]   || 0) + 1;
        layerCounts[lay] = (layerCounts[lay] || 0) + 1;
      }

      var divSwaps = 0;
      var actCategories   = ['tanh', 'relu', 'sigmoid'];
      var layerCategories = [1, 2, 3];

      for (var ai3 = 0; ai3 < actCategories.length; ai3++) {
        var actKey3 = actCategories[ai3];
        if (!actCounts[actKey3]) {
          // Find best candidate from eliminated that has this activation
          for (var ei3 = 0; ei3 < eliminated.length; ei3++) {
            if (eliminated[ei3].cfg.activationFunction === actKey3) {
              var swapScore = eliminated[ei3].compositeScore;
              var scoreThreshold = roundScores[Math.floor(keepCount * SNM_DIVERSITY_SWAP_THRESHOLD)];
              if (scoreThreshold && swapScore > scoreThreshold.compositeScore) {
                // Replace the lowest-scoring promoted candidate
                var worstIdx = promoted.length - 1;
                for (var wi2 = promoted.length - 2; wi2 >= 0; wi2--) {
                  if (promoted[wi2].compositeScore < promoted[worstIdx].compositeScore) { worstIdx = wi2; }
                }
                promoted[worstIdx] = eliminated.splice(ei3, 1)[0];
                divSwaps++;
              }
              break;
            }
          }
        }
      }

      for (var li3 = 0; li3 < layerCategories.length; li3++) {
        var layerKey3 = layerCategories[li3];
        if (!layerCounts[layerKey3]) {
          for (var eli = 0; eli < eliminated.length; eli++) {
            if (eliminated[eli].cfg.numHiddenLayers === layerKey3) {
              var swapScore2 = eliminated[eli].compositeScore;
              var scoreThreshold2 = roundScores[Math.floor(keepCount * SNM_DIVERSITY_SWAP_THRESHOLD)];
              if (scoreThreshold2 && swapScore2 > scoreThreshold2.compositeScore) {
                var worstIdx2 = promoted.length - 1;
                for (var wi3 = promoted.length - 2; wi3 >= 0; wi3--) {
                  if (promoted[wi3].compositeScore < promoted[worstIdx2].compositeScore) { worstIdx2 = wi3; }
                }
                promoted[worstIdx2] = eliminated.splice(eli, 1)[0];
                divSwaps++;
              }
              break;
            }
          }
        }
      }

      // Wildcards: fresh random candidates
      // Generate 5× the needed wildcard count to increase chances of finding unique (unseen) configs
      var wildcards = snmGenerateCandidates(wildCount * SNM_WILDCARD_GEN_MULTIPLIER, makePRNG(rng() * 0xFFFFFF | 0));
      var addedWilds = 0;
      for (var wi = 0; wi < wildcards.length && addedWilds < wildCount; wi++) {
        var wSig = snmConfigSig(wildcards[wi]);
        if (!seenSigs[wSig]) {
          seenSigs[wSig] = true;
          var wScored = await snmEvaluateConfig(wildcards[wi], budget);
          smartDone++;
          totalAnalyses = smartDone;
          totalAnalysesSpan.innerText = smartDone;
          updateProgress(smartDone, totalPossible, smartStartTime);
          if (wScored.compositeScore > snmBestScore) {
            snmBestScore = wScored.compositeScore;
            if (wScored.trainingLosses && wScored.trainingLosses.length) {
              trainingLosses = wScored.trainingLosses.slice();
              validationLosses = (wScored.validationLosses && wScored.validationLosses.length)
                ? wScored.validationLosses.slice() : [];
              updateLossChart();
            }
            if (wScored.matchDist) {
              matchDistribution = wScored.matchDist.slice();
              updateMatchDistChart();
            }
            snmShowInterimBest(wScored);
          }
          promoted.push(wScored);
          addedWilds++;
        }
        await tf.nextFrame();
      }

      // Local refinement from round 3+
      if (round >= SNM_LOCAL_REFINEMENT_START_ROUND) {
        var top10 = promoted.slice(0, Math.min(10, promoted.length));
        for (var ti = 0; ti < top10.length; ti++) {
          var nbrs = snmNeighbors(top10[ti].cfg);
          for (var ni = 0; ni < nbrs.length; ni++) {
            var nSig = snmConfigSig(nbrs[ni]);
            if (!seenSigs[nSig]) {
              seenSigs[nSig] = true;
              var nScored = await snmEvaluateConfig(nbrs[ni], budget);
              smartDone++;
              totalAnalyses = smartDone;
              totalAnalysesSpan.innerText = smartDone;
              updateProgress(smartDone, totalPossible, smartStartTime);
              if (nScored.compositeScore > snmBestScore) {
                snmBestScore = nScored.compositeScore;
                if (nScored.trainingLosses && nScored.trainingLosses.length) {
                  trainingLosses = nScored.trainingLosses.slice();
                  validationLosses = (nScored.validationLosses && nScored.validationLosses.length)
                    ? nScored.validationLosses.slice() : [];
                  updateLossChart();
                }
                if (nScored.matchDist) {

  // Render Top 3 cards from topSettings array
  function skaiRenderTop3(settings) {
    var container = document.getElementById('skaiTop3Cards');
    if (!container) { return; }
    var medals = ['#1','#2','#3'];
    var stabilityThresholds = { high: 0.0015, medium: 0.005 };
    window._skaiTop3Results = [];
    for (var i = 0; i < 3; i++) {
      var r = settings[i];
      if (!r) { continue; }
      window._skaiTop3Results.push({ epochs: r.epochs, decay: r.recencyDecay, skip: r.skipAlpha });
      var scoreEl = document.getElementById('skaiTop3Score' + i);
      var stabEl  = document.getElementById('skaiTop3Stab' + i);
      var whyEl   = document.getElementById('skaiTop3Why' + i);
      if (scoreEl) { scoreEl.textContent = (r.compositeScore || 0).toFixed(4); }
      if (stabEl) {
        var stdDev = r.f1StdDev || 0;
        var stabLabel, stabClass;
        if (stdDev <= stabilityThresholds.high)   { stabLabel = 'High Stability';   stabClass = 'high'; }
        else if (stdDev <= stabilityThresholds.medium) { stabLabel = 'Medium Stability'; stabClass = 'medium'; }
        else                                       { stabLabel = 'Low Stability';    stabClass = 'low'; }
        stabEl.textContent = stabLabel;
        stabEl.className = 'skai-top3-stability ' + stabClass;
      }
      if (whyEl) {
        var reasons = [];
        if (r.avgMatchCount && r.avgMatchCount > 0) {
          reasons.push('Higher average hit score');
        }
        if (r.f1StdDev !== undefined && r.f1StdDev <= stabilityThresholds.medium) {
          reasons.push('More consistent results');
        }
        if (r.compositeScore && r.compositeScore > 0) {
          reasons.push('Strong composite ranking');
        }
        whyEl.textContent = '';
        reasons.slice(0, 3).forEach(function(s) {
          var li = document.createElement('li');
          li.textContent = s;
          whyEl.appendChild(li);
        });
      }
    }
    container.style.display = settings.length ? '' : 'none';
  }

// Build a comma-separated list of values from lo to hi with a given step.
function generateStepList(lo, hi, step, decimals) {
  var result = [];
  var prec = decimals || 0;
  // Multiply by 10^(prec+2) so integer arithmetic handles the loop steps;
  // the extra 2 decimal places buffer against float rounding that would
  // otherwise skip the last value (e.g. 0.30 becoming 0.29999...).
  var scale = Math.pow(10, prec + 2);
  var loS = Math.round(lo * scale);
  var hiS = Math.round(hi * scale);
  var stS = Math.round(step * scale);
  if (stS <= 0) { stS = 1; }
  for (var v = loS; v <= hiS; v += stS) {
    result.push((v / scale).toFixed(prec));
  }
  return result;
}

// Initialise one dual-handle (or single-handle) slider.
// cfg = { id, min, max, step, format, decimals, single }
//   format: 'range'  -> "lo-hi"  (or just "lo" when lo===hi)
//           'list'   -> "lo,lo+step,...,hi"
//           'single' -> just the hi value
function initSlider(cfg) {
  var loEl   = document.getElementById(cfg.id + '_lo');
  var hiEl   = document.getElementById(cfg.id + '_hi');
  var fillEl = document.getElementById(cfg.id + '_fill');
  var dispEl = document.getElementById(cfg.id + '_disp');
  var hidden = document.getElementById(cfg.id);
  if (!hiEl || !hidden) { return; }

  var dec = (cfg.decimals !== undefined) ? cfg.decimals : 0;
  var isSingle = !!cfg.single;

  function refreshFill() {
    var loVal = loEl ? parseFloat(loEl.value) : cfg.min;
    var hiVal = parseFloat(hiEl.value);
    var span  = cfg.max - cfg.min;
    if (!fillEl) { return; }
    if (span === 0) { return; }
    var pLo = ((loVal - cfg.min) / span) * 100;
    var pHi = ((hiVal - cfg.min) / span) * 100;
    fillEl.style.left  = pLo + '%';
    fillEl.style.width = (pHi - pLo) + '%';
    // Raise lo z-index when it's at the maximum so user can drag it left
    if (loEl) {
      loEl.style.zIndex = (parseFloat(loEl.value) >= cfg.max) ? '5' : '2';
    }
  }

  function refreshDisplay() {
    var loVal = loEl ? parseFloat(loEl.value) : cfg.min;
    var hiVal = parseFloat(hiEl.value);
    if (!dispEl) { return; }
    if (isSingle || !loEl || loVal === hiVal) {
      dispEl.textContent = hiVal.toFixed(dec);
    } else {
      dispEl.textContent = loVal.toFixed(dec) + ' – ' + hiVal.toFixed(dec);
    }
  }

  function refreshHidden() {
    var loVal = loEl ? parseFloat(loEl.value) : cfg.min;
    var hiVal = parseFloat(hiEl.value);
    if (isSingle || !loEl) {
      hidden.value = hiVal.toFixed(dec);
      return;
    }
    if (cfg.format === 'range') {
      hidden.value = (loVal === hiVal)
        ? loVal.toFixed(dec)
        : loVal.toFixed(dec) + '-' + hiVal.toFixed(dec);
    } else if (cfg.format === 'list') {
      hidden.value = generateStepList(loVal, hiVal, cfg.step, dec).join(',');
    } else {
      hidden.value = hiVal.toFixed(dec);
    }
    // Notify combos-count updater
    if (typeof updateCombosCount === 'function') {
      // updateCombosCount may not exist on all page variants - ignore gracefully
      try { updateCombosCount(); } catch(e) { /* non-critical UI update */ }
    }
  }

  function onLoChange() {
    if (loEl && hiEl && parseFloat(loEl.value) > parseFloat(hiEl.value)) {
      loEl.value = hiEl.value;
    }
    refreshFill(); refreshDisplay(); refreshHidden();
  }
  function onHiChange() {
    if (loEl && hiEl && parseFloat(hiEl.value) < parseFloat(loEl.value)) {
      hiEl.value = loEl.value;
    }
    refreshFill(); refreshDisplay(); refreshHidden();
  }

  if (loEl) { loEl.addEventListener('input', onLoChange); }
  hiEl.addEventListener('input', onHiChange);

  // Initial render
  refreshFill(); refreshDisplay(); refreshHidden();
}

// ─── Batch-size checkbox pills ───
function initBatchCheckboxes() {
  var row = document.getElementById('batchCheckboxRow');
  if (!row) { return; }
  var hidden = document.getElementById('batchSizes');
  if (!hidden) { return; }

  function updateHidden() {
    var checked = [];
    var chks = row.querySelectorAll('.skai-batch-chk');
    for (var i = 0; i < chks.length; i++) {
      if (chks[i].checked) { checked.push(chks[i].value); }
    }
    hidden.value = checked.length ? checked.join(',') : '4';
    if (typeof updateCombosCount === 'function') {
      // updateCombosCount may not exist on all page variants - ignore gracefully
      try { updateCombosCount(); } catch(e) { /* non-critical UI update */ }
    }
  }

  var pills = row.querySelectorAll('.skai-batch-pill');
  for (var p = 0; p < pills.length; p++) {
    (function (pill) {
      pill.addEventListener('click', function () {
        var chk = pill.querySelector('input[type="checkbox"]');
        if (!chk) { return; }
        chk.checked = !chk.checked;
        pill.classList.toggle('active', chk.checked);
        // Ensure at least one is checked
        var anyChecked = false;
        var allChks = row.querySelectorAll('.skai-batch-chk');
        for (var j = 0; j < allChks.length; j++) {
          if (allChks[j].checked) { anyChecked = true; break; }
        }
        if (!anyChecked) {
          chk.checked = true;
          pill.classList.add('active');
        }
        updateHidden();
      });
    })(pills[p]);
  }
  updateHidden(); // initial sync
}

// ─── Public helper: set slider programmatically (used by presets) ───
function setSliderRange(id, loVal, hiVal) {
  var loEl = document.getElementById(id + '_lo');
  var hiEl = document.getElementById(id + '_hi');
  if (loEl) { loEl.value = loVal; }
  if (hiEl) { hiEl.value = hiVal; }
  // Trigger update via native event dispatch
  ['lo', 'hi'].forEach(function(which) {
    var el = document.getElementById(id + '_' + which);
    if (!el) { return; }
    var evt = new Event('input', { bubbles: true });
    el.dispatchEvent(evt);
  });
}

// ─── Public helper: sync batch checkboxes from a comma string ───
function syncBatchCheckboxes(csvStr) {
  var row = document.getElementById('batchCheckboxRow');
  if (!row) { return; }
  var vals = (csvStr || '4').split(',').map(function (v) { return v.trim(); });
  var pills = row.querySelectorAll('.skai-batch-pill');
  for (var i = 0; i < pills.length; i++) {
    var chk = pills[i].querySelector('input[type="checkbox"]');
    if (!chk) { continue; }
    var isActive = vals.indexOf(chk.value) !== -1;
    chk.checked = isActive;
    pills[i].classList.toggle('active', isActive);
  }
  var hidden = document.getElementById('batchSizes');
  if (hidden) { hidden.value = vals.join(','); }
}
